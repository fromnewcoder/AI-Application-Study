        <tr>

          <td><span class="pill pill-teal">Integration</span></td>

          <td>Judge scores whether every claim in the answer is supported by retrieved context</td>

        </tr>
        <tr>

          <td><span class="pill pill-amber">LLM judge</span></td>

        <tr>

          <td><span class="pill pill-amber">LLM judge</span></td>
          <td>Judge checks if all key aspects of golden answer appear in response</td>

          <td>F1 ≥ 0.70</td>
          <td>Only valid for questions with a definitive short answer. Not for open-ended generation.</td>
        </tr>
        <tr>
          <td><strong>No-answer accuracy</strong></td>

          <td>Run on 50+ examples. Use as final gatekeeper before shipping changes.</td>
        </tr>

  { q: "Which eval type is fastest to run?", opts: ["Integration eval", "Unit eval", "LLM-as-judge", "Human eval"], ans: 1, exp: "Unit evals are isolated and deterministic — they run in milliseconds without LLM calls." },

No-answer accuracy tracks this." },

  { q: "Precision@k measures what?", opts: ["Fraction of relevant docs in top-k results", "Fraction of all relevant docs that appear in top-k", "Number of tokens per retrieved chunk", "Latency of retrieval step"], ans: 0, exp: "Precision@k = relevant docs in top-k / k. Every other eval (automated metric, LLM judge) is validated against it." },
];

let qIdx = 0, score = 0, answered = false;

function renderQuiz() {
  if (qIdx >= questions.length) {
    document.getElementById('quiz-area').innerHTML = `
      <div class="card" style="text-align:center; padding: 2rem;">
        <div class="quiz-score">You scored ${score} / ${questions.length}</div>
        <div class="card-body" style="margin-bottom:1rem">${score === questions.length ? 'Perfect! You nailed it.' : score >= 3 ? 'Good work — review any you missed.' : 'Keep studying — you\'ve got this.'}</div>
        <button class="quiz-next" onclick="resetQuiz()">Retake quiz</button>
      </div>`;
    document.getElementById('quiz-progress').style.width = '100%';
    return;
  }
  const q = questions[qIdx];
  document.getElementById('quiz-progress').style.width = Math.round((qIdx/questions.length)*100) + '%';
  answered = false;
  document.getElementById('quiz-area').innerHTML = `
    <div class="quiz-q">${qIdx+1}. ${q.q}</div>
    <div class="quiz-opts" id="quiz-opts">
      ${q.opts.map((o,i) => `<button class="quiz-opt" onclick="selectOpt(${i})">${o}</button>`).join('')}
    </div>
    <div id="quiz-fb"></div>
  `;
}

function selectOpt(i) {
  if (answered) return;
  answered = true;
  const q = questions[qIdx];
  const opts = document.querySelectorAll('.quiz-opt');
  opts[i].classList.add(i === q.ans ? 'correct' : 'wrong');
  if (i !== q.ans) opts[q.ans].classList.add('reveal');
  if (i === q.ans) score++;
  document.getElementById('quiz-fb').innerHTML = `
    <div class="quiz-feedback ${i === q.ans ? 'correct' : 'wrong'}">${i === q.ans ? '✓ Correct — ' : '✗ Not quite — '}${q.exp}</div>
    <button class="quiz-next" onclick="advanceQuiz()">${qIdx+1 < questions.length ? 'Next question' : 'See results'}</button>
  `;
}

function advanceQuiz() { qIdx++; renderQuiz(); }
function resetQuiz() { qIdx = 0; score = 0; renderQuiz(); }
renderQuiz();
</script>
